<script>
    const texts = [
    "سمیه اشگوریان - وکیل پایه یک دادگستری",
    "مشاوره تخصصی در دعاوی ملکی",
    "وکیل دعاوی خانواده، کیفری و حقوقی"
    ];

    let textIndex = 0;
    let charIndex = 0;
    let isDeleting = false;
    let typingSpeed = 100; // سرعت تایپ
    let deletingSpeed = 50; // سرعت حذف
    let delayBetweenTexts = 1500; // تأخیر بین نمایش جملات

    function typeEffect() {
        let typingText = document.getElementById("typing-text");
    let currentText = texts[textIndex];

    if (isDeleting) {
        charIndex--; // کم کردن متن در حالت حذف
        } else {
        charIndex++; // اضافه کردن متن در حالت تایپ
        }

    typingText.textContent = currentText.substring(0, charIndex);

    if (!isDeleting && charIndex === currentText.length) {
        isDeleting = true; // شروع حذف پس از تکمیل تایپ
    setTimeout(typeEffect, delayBetweenTexts);
        } else if (isDeleting && charIndex === 0) {
        isDeleting = false; // شروع تایپ متن بعدی
    textIndex = (textIndex + 1) % texts.length;
    setTimeout(typeEffect, typingSpeed);
        } else {
        setTimeout(typeEffect, isDeleting ? deletingSpeed : typingSpeed);
        }
    }

    // شروع تایپ بعد از لود صفحه
    window.onload = function () {
        document.getElementById("typing-text").textContent = "";
    setTimeout(typeEffect, 1000);
    };
</script>
